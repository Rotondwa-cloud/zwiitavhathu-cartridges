const sqlite3 = require('sqlite3').verbose();
const fs = require('fs');

const db = new sqlite3.Database('./database.db');

const sql = fs.readFileSync('import_cartridges.sql', 'utf8');

db.exec(sql, (err) => {
  if (err) {
    console.error("❌ Import failed:", err.message);
  } else {
    console.log("✅ Cartridges imported successfully from SQL file!");
  }
  db.close();
});
